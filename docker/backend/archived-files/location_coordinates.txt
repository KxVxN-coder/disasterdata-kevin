import aiohttp
import asyncio

API_KEY="ZIYuaacyfnN4Y17OIqaUMT7YBZWz3M6usplChpoIK3g"
geocode_url="https://geocode.search.hereapi.com/v1/geocode"

async def geocode_location(address):
    params = {'q': address, 'apiKey': API_KEY}
    
    async with aiohttp.ClientSession() as session:
        async with session.get(geocode_url, params=params) as response:
            if response.status == 200:
                data = await response.json()
                if data.get('items'):
                    location = data['items'][0]['position']
                    print(f"Address: {address} - Lat: {location['lat']}, Lon: {location['lng']}")
                else:
                    print(f"Address: {address} - No geocoding result found.")
            else:
                print(f"Error geocoding {address}: {response.status}")

def main():
    address = "1600 Pennsylvania Ave, Washington, DC"  # Replace with the address you want to geocode
    asyncio.run(geocode_location(address))

if __name__ == "__main__":
    main()
